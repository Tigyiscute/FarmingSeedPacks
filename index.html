01

<iframe> — Inline Frame

The native HTML element for embedding another webpage inside yours. The browser renders the target URL in a sandboxed viewport.

✓ Native HTML ✓ Sandboxed ⚠ Blocked by X-Frame-Options 
<!-- Basic iframe embed -->
<iframe
  src="https://gag.gg"
  width="100%"
  height="600"
  frameborder="0"
  loading="lazy"
  title="gag.gg"
  allow="fullscreen"
  sandbox="allow-scripts allow-same-origin allow-forms allow-popups"
></iframe>
/* Responsive iframe with 16:9 ratio */
.iframe-wrapper {
  position: relative;
  width: 100%;
  padding-bottom: 56.25%; /* 16:9 */
  height: 0;
  overflow: hidden;
}

.iframe-wrapper iframe {
  position: absolute;
  top: 0; left: 0;
  width: 100%;
  height: 100%;
  border: none;
}

02

Fetch via Server Proxy

Route the request through your own server to bypass X-Frame-Options and CORS. Your backend fetches the HTML, strips/rewrites asset URLs, and serves it to your front end.

✓ Bypasses X-Frame-Options ⚠ Requires server ✗ TOS risk 
// Express.js reverse proxy — server.js
const express = require('express');
const { createProxyMiddleware } = require('http-proxy-middleware');

const app = express();

app.use('/proxy', createProxyMiddleware({
  target: 'https://gag.gg',
  changeOrigin: true,
  pathRewrite: { '^/proxy': '' },
  on: {
    proxyRes(proxyRes) {
      // Strip framing-prevention headers
      delete proxyRes.headers['x-frame-options'];
      delete proxyRes.headers['content-security-policy'];
    }
  }
}));

app.listen(3000);
// Client-side: load proxy URL into iframe
const frame = document.createElement('iframe');
frame.src = '/proxy/'; // hits your local proxy
frame.style.width  = '100%';
frame.style.height = '600px';
frame.style.border = 'none';

document.getElementById('container').appendChild(frame);
⚠️ Proxying a third-party website without permission may violate their Terms of Service and applicable law. Use only for sites you own or have authorisation for.
03

window.open() — Popup / New Tab

Launch the URL in a controlled browser window. You can size and position it like a modal. Not technically "on the page", but creates the appearance of an inline experience.

✓ Always works ⚠ Blocked by popup blockers → New window 
/**
 * Opens gag.gg in a centred popup window.
 * Must be called from a user-gesture (click, keydown)
 * to avoid popup blockers.
 */
function openGagGG() {
  const w = 1024, h = 768;
  const left = (screen.width  - w) / 2;
  const top  = (screen.height - h) / 2;

  window.open(
    'https://gag.gg',
    'gag_window',
    `width=${w},height=${h},left=${left},top=${top},` +
    'toolbar=no,menubar=no,scrollbars=yes,resizable=yes'
  );
}

// Bind to a button
document
  .getElementById('open-btn')
  .addEventListener('click', openGagGG);

04

window.location / <a> redirect

Navigate the current tab to the URL — either via a JavaScript redirect or a plain anchor tag. Simplest method; fully replaces your page.

✓ Zero restrictions ✗ Leaves your page 
// Hard redirect — replaces browser history entry
window.location.replace('https://gag.gg');

// Soft redirect — adds history entry (back button works)
window.location.href = 'https://gag.gg';

// Delayed redirect (e.g. after 3 seconds)
setTimeout(() => {
  window.location.href = 'https://gag.gg';
}, 3000);
<!-- Opens in same tab -->
<a href="https://gag.gg">Go to gag.gg</a>

<!-- Opens in new tab -->
<a
  href="https://gag.gg"
  target="_blank"
  rel="noopener noreferrer"
>Open gag.gg in new tab</a>
<!-- Auto-redirect after 5 seconds (in <head>) -->
<meta
  http-equiv="refresh"
  content="5; url=https://gag.gg"
/>

<!-- Instant redirect -->
<meta
  http-equiv="refresh"
  content="0; url=https://gag.gg"
/>
05

<object> Element

An older HTML element that can embed external resources including web pages. Browsers treat it similarly to an iframe — same X-Frame-Options restrictions apply.

↔ Like iframe ⚠ Same frame restrictions ⚠ Limited browser support 
<object
  data="https://gag.gg"
  type="text/html"
  width="100%"
  height="600"
  style="border:none;"
>
  <!-- Fallback if object is not supported -->
  <a href="https://gag.gg">Visit gag.gg</a>
</object>
06

<embed> Element

A void HTML element primarily for plugins (Flash, PDF). Modern browsers may load HTML URLs in it, though behaviour is inconsistent and largely unsupported for web pages.

✗ Not reliable for HTML → Best for PDFs 
<!-- Note: <embed> is unreliable for embedding web pages.
     Prefer <iframe> for HTML and <embed> for PDFs only. -->

<embed
  src="https://gag.gg"
  type="text/html"
  width="100%"
  height="600"
/>

<!-- Works well for PDFs -->
<embed
  src="https://example.com/doc.pdf"
  type="application/pdf"
  width="100%"
  height="600"
/>
